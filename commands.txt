#!/bin/sh

# Goals

# equivalent
karyn add --bucket name --key name --value value
karyn bucket/name value
karyn add --bucket name --key name --value value --content-type application/json


# equivalent:
karyn list --bucket name --key name
karyn list -b name -k name
karyn list bucket/name

# list all keys in bucket
karyn list --bucket name

# search all buckets, equivalent
karyn list --bucket _ --key name
karyn list --key name
karyn list _/name

# delete key in bucket
karyn list --bucket name --key --delete

# delete key of name in any bucket
karyn list --bucket _ --key --delete
karyn list --key --delete

# delete all keys in bucket
karyn list --bucket name --delete

# delete all keys in all buckets
karyn list --bucket _ --delete

# show all links in bucket/key
karyn list --bucket name --key name --links

# show all links key, look in all buckets
karyn list --bucket _ --key name --links
karyn list --key name --links
